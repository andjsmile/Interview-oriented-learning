### 2.ZOOM

1.输出一个字符串(200)的长度,将大写字母改为小写，小写字母改为大写
  输出修改后的字符串

ac-0.2
```cpp
int main()
{
    string str;
    cin>>str;
    int len=str.size();
    for(int i=0;i<len;++i)
    {
        if(str[i]>='a' && str[i]<='z')
        {
            str[i]=toupper(str[i]);
        }
        else if(str[i>='A' && str[i]<='Z'])
        {
            str[i]=tolower(str[i]);
        }
    }
    cout<<str<<endl;
    return 0;
}
```
